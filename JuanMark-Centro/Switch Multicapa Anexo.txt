enable
configure terminal
hostname MSW_ANEXO

! ========================================
!  CONFIGURACIÓN BASE Y ACCESO SSH 
! ========================================
service password-encryption
ip domain-name juanmark.com
username admin secret cisco
crypto key generate rsa
1024
line vty 0 4
 transport input ssh
 login local
exit
line con 0
 logging synchronous
exit

! =========================
!  CONFIGURACIÓN VTP 
! =========================
vtp domain juanmark
vtp version 2
vtp mode client

! =========================================
!  AJUSTES GLOBALES Y SPANNING-TREE 
! =========================================
no ip cef
no ipv6 cef
spanning-tree mode pvst
spanning-tree extend system-id

! ===========================
!  ENLACES TRONCALES LAN 
! ===========================

! Port-channel 1: Troncal Lógico a MSW_Principal
interface Port-channel1
 switchport mode trunk
 ip dhcp snooping trust
 ip arp inspection trust
 no shutdown
exit

! G1/0/12-13: Puertos Físicos para EtherChannel
interface range GigabitEthernet1/0/12-13
 switchport mode trunk
 channel-protocol pagp
 channel-group 1 mode desirable
 ip dhcp snooping trust
 ip arp inspection trust
 no shutdown
exit

! G1/0/3: Troncal a SW_Anexo
interface GigabitEthernet1/0/3
 switchport mode trunk
 ip dhcp snooping trust
 ip arp inspection trust
 no shutdown
exit
