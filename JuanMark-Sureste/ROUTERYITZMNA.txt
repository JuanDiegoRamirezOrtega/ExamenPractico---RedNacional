enable
configure terminal
hostname Router_Itzamna

! --- CONFIGURACIÓN BASE Y SSH (Punto 1) ---
service password-encryption
enable secret cisco
username admin secret cisco
ip domain-name juanmark.com
crypto key generate rsa modulus 1024
line vty 0 15
 transport input ssh
 login local
exit

! --- PASO 1: Encender la Interfaz Física ---
interface g0/0/0
 no shutdown
 exit

! --- PASO 2: Configurar Subinterfaces y HSRP (Punto 5) ---
interface g0/0/0.65
 encapsulation dot1Q 65
 ip address 148.60.65.2 255.255.255.0
 standby 65 ip 148.60.65.1
 standby 65 priority 150
 standby 65 preempt
 exit
interface g0/0/0.66
 encapsulation dot1Q 66
 ip address 148.60.66.2 255.255.255.0
 standby 66 ip 148.60.66.1
 standby 66 priority 150
 standby 66 preempt
 exit
interface g0/0/0.67
 encapsulation dot1Q 67
 ip address 148.60.67.2 255.255.255.0
 standby 67 ip 148.60.67.1
 standby 67 priority 150
 standby 67 preempt
 exit
interface g0/0/0.68
 encapsulation dot1Q 68
 ip address 148.60.68.2 255.255.255.0
 standby 68 ip 148.60.68.1
 standby 68 priority 150
 standby 68 preempt
 exit

! --- PASO 3: Configurar el Servidor DHCP (Punto 5) ---
ip dhcp excluded-address 148.60.65.1 148.60.65.3
ip dhcp pool VENTAS
 network 148.60.65.0 255.255.255.0
 default-router 148.60.65.1
 dns-server 8.8.8.8
 exit
ip dhcp excluded-address 148.60.66.1 148.60.66.3
ip dhcp pool TI
 network 148.60.66.0 255.255.255.0
 default-router 148.60.66.1
 dns-server 8.8.8.8
 exit
ip dhcp excluded-address 148.60.67.1 148.60.67.3
ip dhcp pool COMPRAS
 network 148.60.67.0 255.255.255.0
 default-router 148.60.67.1
 dns-server 8.8.8.8
 exit
ip dhcp excluded-address 148.60.68.1 148.60.68.3
ip dhcp pool MKT
 network 148.60.68.0 255.255.255.0
 default-router 148.60.68.1
 dns-server 8.8.8.8
 exit

! --- PASO 4: Enrutamiento RIPv2 (Punto 3) ---
router rip
 version 2
 no auto-summary
 network 148.60.0.0
 passive-interface g0/0/0.65
 passive-interface g0/0/0.66
 passive-interface g0/0/0.67
 passive-interface g0/0/0.68
 exit
