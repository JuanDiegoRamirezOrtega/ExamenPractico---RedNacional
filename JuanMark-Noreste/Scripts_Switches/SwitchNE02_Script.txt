SwitchNE02 (conectado directamente a LaVaca)
enable
configure terminal
hostname SwitchNE02
ip domain-name juanmark.com


enable secret cisco
service password-encryption
username admin secret cisco123

crypto key generate rsa

ip ssh version 2

line vty 0 4
login local
transport input ssh
exit


vtp mode client
vtp domain juanmark
vtp password cisco
vtp version 2


spanning-tree mode rapid-pvst

spanning-tree vlan 99,100 root primary


spanning-tree vlan 97,98 root secondary

! --- . ENLACE HACIA LAVACA ---
interface fa0/1
description Link_to_Router_Lavaca
switchport trunk encapsulation dot1q
switchport mode trunk
switchport nonegotiate
switchport trunk allowed vlan 97-100
no shutdown
exit

---  ENLACE HACIA SWITCHNE01  ---
interface g0/1
description Link_to_SwitchNE01
switchport trunk encapsulation dot1q
switchport mode trunk
switchport nonegotiate
switchport trunk allowed vlan 97-100
no shutdown
exit

 ---  ENLACE HACIA SWITCHNE02  ---

description Link_to_SwitchLaSilla
switchport trunk encapsulation dot1q
switchport mode trunk
switchport nonegotiate
switchport trunk allowed vlan 97-100
no shutdown
exit
